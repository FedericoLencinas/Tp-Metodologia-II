//Como usuario, necesito crear una aplicación que me permita ingresar los datos de las 
//familias de una manzana, para mostrar y determinar que familia tiene más mujeres,
//la cantidad total de hombres y mujeres, y el porcentaje de casas propias.
double canthombres = 0, cantmujeres = 0, porcentajecasapropia = 0, casapropia = 0, maxmujeres = 0;
int i;

int mujeres;
Console.WriteLine("Ingrese la cantidad de familias (por lo menos 5): ");
double n=int.Parse(Console.ReadLine());

for( i=1; i<=n; i++)
{
    Console.WriteLine($"Ingrese la cantidad de mujeres de la familia {i}:");
    mujeres=int.Parse(Console.ReadLine());
    cantmujeres=cantmujeres+mujeres;
    if (i == 1 & mujeres > maxmujeres)
    {
        maxmujeres = mujeres;
        mujeres = i;
    }
    Console.WriteLine($"Ingrese la cantidad de hombres de la familia {i}");
    int hombres =int.Parse(Console.ReadLine());
    canthombres = canthombres + hombres;
    Console.WriteLine("Ingrese si posee casa propia \n1- Si\n2- No: ");
    casapropia=int.Parse(Console.ReadLine());
    if (casapropia == 1)
    {
        casapropia++;
    } 
    
    
}
porcentajecasapropia = (casapropia * 100) / n;
Console.WriteLine($"La cantidad total de hombres es de: {canthombres} ");
Console.WriteLine($"La cantidad total de mujeres es de: {cantmujeres} ");
Console.WriteLine($"El porcentaje de casas propias es de: %{porcentajecasapropia}");
Console.WriteLine($"La familia con mas mujeres es la familia: {i}");
